WITH ContactRanked AS (
    SELECT 
        T4."CardCode", 
        T4."Name" AS "Contact Person",
        ROW_NUMBER() OVER (PARTITION BY T4."CardCode" ORDER BY T4."CntctCode") AS "RowNum"
    FROM PRD.OCPR T4
),
SO_LineCount AS (
    SELECT 
        T3."DocEntry", 
        COUNT(*) AS "Total Lines in SO"
    FROM PRD.RDR1 T0
    LEFT JOIN PRD.ORDR T3 ON T0."DocEntry" = T3."DocEntry"
    GROUP BY T3."DocEntry"
)
SELECT DISTINCT
    -- üÜï Gi·ªØ nguy√™n c√°c c·ªôt hi·ªán c√≥, KH√îNG thay ƒë·ªïi th·ª© t·ª±
    T2."CardCode" AS "Customer Code",
    T2."CardName" AS "Customer Name",
    T0."ItemCode" AS "Item Code",
    T1."ItemName" AS "Item Name",
    T1."SalUnitMsr" AS "Sale UoM",
    T0."Quantity" AS "Quantity Ordered",
    T0."LineTotal" AS "Total Amount without VAT",
    ROUND(T0."LineTotal" * (1 + (T0."VatPrcnt" / 100)), 2) AS "Total Amount with VAT",
    CASE
        WHEN T0."Quantity" > 0 
        THEN ROUND((T0."LineTotal" * (1 + (T0."VatPrcnt" / 100))) / T0."Quantity", 2)
        ELSE NULL
    END AS "Price per Pack with VAT",
    T3."DocNum" AS "Sales Order Number",
    TO_VARCHAR(T3."DocDate", 'DD/MM/YYYY') AS "Sales Order Date",
    CR."Contact Person",
    T5."SlpName" AS "Sales Employee Name",
    QUARTER(T3."DocDate") AS "Quarter",
    TO_VARCHAR(T3."DocDate", 'YYYY-MM') AS "Year-Month",
    CASE 
        WHEN T3."DocStatus" = 'O' THEN 'Ch∆∞a ho√†n th√†nh'
        WHEN T3."DocStatus" = 'C' THEN 'ƒê√£ ho√†n th√†nh'
        ELSE 'Kh√¥ng x√°c ƒë·ªãnh'
    END AS "Sales Order Status",
    T3."CANCELED" AS "Is Canceled",
    INV."DocNum" AS "A/R Invoice Number",
    TO_VARCHAR(INV."DocDate", 'DD/MM/YYYY') AS "A/R Invoice Date",
    INV."DocTotal" AS "Invoice Total Amount",
    DLV."DocNum" AS "Delivery Number",
    TO_VARCHAR(DLV."DocDate", 'DD/MM/YYYY') AS "Delivery Date",
    DLV."DocTotal" AS "Delivery Total Amount",

    -- üÜï **Th√™m 3 c·ªôt v·ªÅ gi·∫£m gi√° ·ªü cu·ªëi b·∫£ng**
    CASE 
        WHEN T3."DiscPrcnt" > 0 OR T3."DiscSum" > 0 THEN 'C√≥ gi·∫£m gi√°' 
        ELSE 'Kh√¥ng'
    END AS "Has Discount",  -- Ki·ªÉm tra c√≥ gi·∫£m gi√° hay kh√¥ng

    T3."DiscSum" AS "Discount Amount",  -- T·ªïng ti·ªÅn gi·∫£m gi√° c·ªßa ƒë∆°n h√†ng

    ROUND(T3."DiscSum" / SOLC."Total Lines in SO", 2) AS "Discount Trung B√¨nh"  -- Ph√¢n b·ªï gi·∫£m gi√° tr√™n m·ªói d√≤ng s·∫£n ph·∫©m

FROM PRD.RDR1 T0  
LEFT JOIN PRD.OITM T1 ON T0."ItemCode" = T1."ItemCode"
LEFT JOIN PRD.ORDR T3 ON T0."DocEntry" = T3."DocEntry"
LEFT JOIN PRD.OCRD T2 ON T3."CardCode" = T2."CardCode"
LEFT JOIN ContactRanked CR ON T2."CardCode" = CR."CardCode" AND CR."RowNum" = 1
LEFT JOIN PRD.OSLP T5 ON T3."SlpCode" = T5."SlpCode"

-- üìå **Th√™m th√¥ng tin h√≥a ƒë∆°n**
LEFT JOIN PRD.DLN1 DLN1 ON T3."DocEntry" = DLN1."BaseEntry" AND DLN1."BaseType" = 17  
LEFT JOIN PRD.ODLN DLV ON DLN1."DocEntry" = DLV."DocEntry"  
LEFT JOIN PRD.INV1 INV1 ON DLV."DocEntry" = INV1."BaseEntry" AND INV1."BaseType" = 15  
LEFT JOIN PRD.OINV INV ON INV1."DocEntry" = INV."DocEntry"

-- üìå **Join b·∫£ng ƒë·∫øm s·ªë d√≤ng c·ªßa SO**
LEFT JOIN SO_LineCount SOLC ON T3."DocEntry" = SOLC."DocEntry"

WHERE 
    -- üìå **L·ªçc theo danh s√°ch c√¥ng ty**
    T2."CardCode" IN ('211047', '211057', '211058', '211059', '211060', '211061', '216012', '211062', '216014', '216015', '216016', '216017', '216018', '216019', '216020', '216021', '211063', '216022', '216023', '216024', '216025', '211064', '216026', '216027', '216028', '211065', '211066', '216029', '211067', '211068', '211069', '211070', '211071', '216030', '211072', '216032', '216013', '216031', '216033', '211073', '211074', '211075', '211076', '211077', '211078', '211079', '211080', '216034', '211081', '211082', '211084', '211083', '211085', '211086', '211087', '211088', '211089', '211090', '211091', '211092', '211129', '211048', '211050', '211051', '211052', '211053', '216009', '216007', '211046', '211044', '211105', '211106', '211107', '211108', '211049', '211055', '216036', '216008', '216037', '216038', '216039', '216040', '216041', '216042', '216043', '216044', '216045', '216046', '216047', '216048', '216049', '216050', '216051', '216052', '216053', '216054', '216055', '216056', '216057', '216058', '216059', '216060', '216061', '216062', '216063', '216064', '216065', '216066', '216084', '216083', '216082', '216081', '216080', '216079', '216077', '216078', '216076', '216075', '216074', '216073', '216072', '216071', '216070', '216069', '216068', '216067', '211119', '211120', '211056', '211093', '211094', '211095', '211097', '211098', '211099', '211100', '211102', '211103', '211104', '211110', '211111', '211112', '211113', '211114', '211115', '211116', '211117', '211118', '211121', '211109', '211123', '211124', '211125', '211126', '211127', '216085', '216086', '211128', '211132', '211134')

    -- üìå **Lo·∫°i b·ªè A/R Invoice b·ªã h·ªßy**
    AND (INV."DocStatus" <> 'C' OR INV."CANCELED" NOT IN ('C', 'Y'))

ORDER BY 
    T2."CardCode", 
    T3."DocNum", 
    "Year-Month" DESC,  
    "Quarter" DESC,  
    T0."ItemCode";
